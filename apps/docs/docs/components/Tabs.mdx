## Code

```tsx
import Tabs from "@repo/ui/Tabs";
import { useState } from "react";
import { CircleX } from "lucide-react";

function TabsWrapper() {
  const [value, setValue] = useState(1);

  return (
    <>
       <Tabs value={value} onChange={(v) => setValue(v)}>
        <Tabs.Item>
          <span>عنوان</span>
          <CircleX size={16} />
        </Tabs.Item>
        <Tabs.Item>
          <span>عنوان</span>
          <CircleX size={16} />
        </Tabs.Item>
        <Tabs.Item>
          <span>عنوان</span>
          <CircleX size={16} />
        </Tabs.Item>
      </Tabs>
    </>
  );
}
````

## Props

### `Tabs` component

| Prop       | Type                         | Default |
| ---------- | ---------------------------- | ------- |
| `value`    | `number`                     | -       |
| `onChange` | `(newValue: number) => void` | -       |
| `children` | `ReactNode`                  | -       |

### `Tabs.Item` component

| Prop        | Type         | Default |
| ----------- | ------------ | ------- |
| `active`    | `boolean`    | `false` |
| `onClick`   | `() => void` | -       |
| `className` | `string`     | -       |
| `children`  | `ReactNode`  | -       |
