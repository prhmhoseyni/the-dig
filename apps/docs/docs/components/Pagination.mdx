## Code

```tsx
"use client";

import Pagination from "@repo/ui/Pagination";
import { useState } from "react";

const [page, setPage] = useState(1);

return (
  <Pagination
    total={10}
    page={page}
    onChange={(p) => setPage(p)}
  />
);
````

## Installation

```bash
npx the-dig@latest add Pagination
```
