# IconButton

## Preview

import { Activity } from "lucide-react";
import IconButton from "@repo/ui/IconButton";
import Container from "./Container";
import RedirectLink from "../../src/components/RedirectLink";

<Container>
  <IconButton>
    <Activity size={18} />
  </IconButton>
</Container>

```tsx
import IconButton from "@repo/ui/IconButton";
import { Activity } from "lucide-react";

<IconButton>
  <Activity size={18} />
</IconButton>;
```

## Installation

```bash
npx the-dig@latest add IconButton
```

## Props

| Prop        | Type                                                                          | Default                                                                                                         |
| ----------- | ----------------------------------------------------------------------------- | --------------------------------------------------------------------------------------------------------------- |
| `variant`   | `"contained"` \| `"tinted"` \| `"outlined"`                                   | `"contained"`                                                                                                   |
| `color`     | `"brand"` \| `"info"` \| `"success"` \| `"warning"` \| `"danger"` \| `"gray"` | `"brand"`                                                                                                       |
| `size`      | `"xs"` \| `"sm"` \| `"md"` \| `"lg"` \| `"xl"`                                | `"md"`                                                                                                          |
| `isLoading` | `boolean`                                                                     | `false`                                                                                                         |
| Others      | `ButtonHTMLAttributes<HTMLButtonElement>` (e.g. `onClick`, `disabled`, etc.)  | <RedirectLink href="https://developer.mozilla.org/en-US/docs/Web/API/HTMLButtonElement">Document</RedirectLink> |
