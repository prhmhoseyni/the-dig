# Menu

## Preview

import Menu from "@repo/ui/Menu";
import MenuPreview from "../../src/components/MenuPreview";
import Container from "./Container";

<Container>
  <MenuPreview />
</Container>

```tsx
import { useRef, useState } from "react";
import Menu from "@repo/ui/Menu";
import Button from "@repo/ui/Button";

const [open, setOpen] = useState(false);
const anchor = useRef<HTMLButtonElement>(null);

<Button ref={anchor} onClick={() => setOpen((prev) => !prev)}>
  روم کلیک کن تا نشونت بدم
</Button>

<Menu anchor={anchor.current} open={open} onClose={() => setOpen(false)}>
  <Menu.Item onClick={() => console.log("Item 1 clicked")}>
    گزینه اول
  </Menu.Item>
  <Menu.Item onClick={() => console.log("Item 2 clicked")}>
    گزینه دوم
  </Menu.Item>
  <Menu.Item onClick={() => console.log("Item 3 clicked")}>
    گزینه سوم
  </Menu.Item>
</Menu>
```
